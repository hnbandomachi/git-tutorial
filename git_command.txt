git show ID_commit:	Xem thay đổi của commit
git diff:			Chỉ xem nội dung của các file có modified\

working directory:	Màu đỏ
	|
staging area:		Màu xanh
	|
git repository:		Màu trắng


git checkout --file: bỏ modified của file, thay đổi nội dung, nhưng không đưa xuống untrack file
git restore --staged file: chuyển đỏ, xanh -> đỏ(hoặc untrack unless commit) không thay đổi nội dung (not recommend)
git reset file: chuyển từ xanh -> đỏ


git checkout -b name_branch:	
	checkout: chuyển sang nhánh đó làm việc
	-b: tạo branch mới
git merge branch_name: merge branch_name vào branch đang đứng
git branch -D branch_name: đứng ở master và xóa branch_name


git reset --soft <ID_commit>: mở hộp (trắng -> xanh)
git reset --mixed <ID_commit>: trắng -> đỏ
git reset --hard <ID_commit>: trắng -> trắng
